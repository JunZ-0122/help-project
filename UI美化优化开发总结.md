### 3. 动态颜色系统
```typescript
// 状态颜色动态计算
private getStatusColor(status: string): string {
  switch (status) {
    case 'assigned': return '#3b82f6';
    case 'in-progress': return '#f59e0b';
    case 'completed': return '#10b981';
    // ...
  }
}
```

  radius: 8,
