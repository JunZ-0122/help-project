### 3. 常见错误快速修复

#### 错误：Object literal must correspond to interface
```typescript
// ❌ 错误
const data = { name: 'test', age: 20 };

// ✅ 正确
interface UserData {
  name: string;
  age: number;
}
const data: UserData = { name: 'test', age: 20 };
```

#### 错误：'this' cannot be used in static methods
```typescript
// ❌ 错误
static method() {
  return this.otherMethod();
}

// ✅ 正确
static method() {
  return ClassName.otherMethod();
}
```

#### 错误：Indexed signatures are not supported
```typescript
// ❌ 错误
interface Data {
  [key: string]: number;
}

// ✅ 正确
interface Data {
  field1: number;
  field2: number;
  field3: number;
}
```

#### 错误：Throw statements can only throw Error
```typescript
// ❌ 错误
catch (error) {
  throw error.message;
}

// ✅ 正确
catch (err) {
  const error = err as Error;
  throw new Error(error.message || '操作失败');
}
```

#### 错误：Destructuring declarations not supported
```typescript
// ❌ 错误
const { name, age } = user;

// ✅ 正确
const name = user.name;
const age = user.age;
```

#### 错误：Dynamic import is not supported
```typescript
// ❌ 错误
const module = await import('./Service');

// ✅ 正确
import { Service } from './Service';
```

## ArkTS 代码结构规范

### 正确的文件结构
```typescript
// 1. 导入
import { Type1 } from './module1';

// 2. 接口定义（模块级别）
interface DataType {
  field: string;
}

// 3. 类定义
export class MyClass {
  static method() { ... }
}
```

### 错误的结构
```typescript
// ❌ 不要这样做
export class MyClass {
  static method1() { ... }
  
  interface DataType { ... }  // 接口不能在类内部
  
  static method2() { ... }
}
```
